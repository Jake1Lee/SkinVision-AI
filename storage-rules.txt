// Firebase Storage Security Rules for SkinVision-AI

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Scan images - users can upload/read their own images only
    match /scan-images/{userId}_{timestamp}_{filename} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // PDF reports - users can upload/read their own PDFs only
    match /scan-pdfs/{userId}_{scanId}_{timestamp}_{filename} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Deny all other access
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
